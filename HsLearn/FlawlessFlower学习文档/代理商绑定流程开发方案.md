## 1. 功能概述

### 1.1 业务场景

```
┌──────────┐    ① 输入绑定码/扫二维码     ┌──────────────┐
│  机台     │ ──────────────────────────→ │  云端服务器    │
│ (Kiosk)  │ ←────────────────────────── │              │
└──────────┘    ② 验证通过→下发代理商信息  └──────────────┘
     │                                         │
     │  ③ 保存代理商信息到本地                   │
     │     机台归属该代理商                       │
     ▼                                         ▼
┌──────────┐                           ┌──────────────┐
│ 重启后    │                           │ 代理商后台     │
│ 自动检查  │                           │ 可看到机台列表  │
│ 已绑定    │                           │ 营收数据等     │
└──────────┘                           └──────────────┘
```

### 1.2 约束条件

- 机台在 **未绑定代理商** 前不能进入主界面
- 绑定码由代理商在管理后台生成，格式为 **Agent 绑定二维码** 或 **16位绑定码**
- 绑定信息本地持久化（JSON + AES 加密），断网也可检查已绑定状态
- 已在后台管理中可查看/解绑

```csharp
using System;

/// <summary>代理商信息</summary>
[Serializable]
public class AgentInfo
{
    public string agentId;         // 代理商ID
    public string agentName;       // 代理商名称
    public string agentPhone;      // 联系电话
    public string bindTime;        // 绑定时间 yyyy-MM-dd HH:mm:ss
    public string expiredTime;     // 授权到期时间 (空=永久)
    public int bindMode;           // 0=永久 1=租用
}

/// <summary>绑定请求</summary>
[Serializable]
public class AgentBindRequest
{
    public string machineCode;     // 机台唯一标识
    public string bindCode;        // 16位绑定码
    public string macId;           // 网卡MAC
}

/// <summary>绑定响应</summary>
[Serializable]
public class AgentBindResponse
{
    public int code;               // 0=成功, 1001=无效绑定码, 1002=已过期, 1003=已被绑定
    public string message;
    public AgentInfo data;
}

/// <summary>解绑响应</summary>
[Serializable]
public class AgentUnbindResponse
{
    public int code;
    public string message;
}
```

```csharp
// Assets/Scripts/Game/AgentBindModel.cs
using Newtonsoft.Json;
using System.IO;
using UnityEngine;

/// <summary>
/// 代理商绑定数据模型 —— 使用框架 BaseManager 单例模式
/// </summary>
public class AgentBindModel : BaseManager<AgentBindModel>
{
    private const string BIND_FILE = "agent_bind.json";
    private const string LOCAL_AES_KEY = "AgentBind_LocalKey_32Bytes!!!!!!";

    public AgentInfo CurrentAgent { get; private set; }
    public bool IsBound => CurrentAgent != null
        && !string.IsNullOrEmpty(CurrentAgent.agentId);

    public void Init()
    {
        LoadFromLocal();
    }

    /// <summary>从本地文件读取绑定信息</summary>
    public void LoadFromLocal()
    {
        string path = Path.Combine(Application.persistentDataPath, BIND_FILE);
        if (!File.Exists(path))
        {
            Debug.Log("[AgentBindModel] 本地无绑定记录，机台未绑定");
            return;
        }

        try
        {
            string encrypted = File.ReadAllText(path);
            // 使用统一 AES 解密
            string json = AESScript.Decrypt(encrypted, LOCAL_AES_KEY);
            CurrentAgent = JsonConvert.DeserializeObject<AgentInfo>(json);
            Debug.Log($"[AgentBindModel] 读取绑定记录: {CurrentAgent.agentName}");
        }
        catch (System.Exception e)
        {
            Debug.LogError($"[AgentBindModel] 读取绑定文件失败: {e.Message}");
            CurrentAgent = null;
        }
    }

    /// <summary>保存绑定信息到本地 (AES加密)</summary>
    public void SaveToLocal(AgentInfo agent)
    {
        CurrentAgent = agent;
        string json = JsonConvert.SerializeObject(agent);
        string encrypted = AESScript.Encrypt(json, LOCAL_AES_KEY);
        string path = Path.Combine(Application.persistentDataPath, BIND_FILE);
        File.WriteAllText(path, encrypted);
        Debug.Log($"[AgentBindModel] 绑定信息已保存: {agent.agentName}");
    }

    /// <summary>清除本地绑定信息 (解绑)</summary>
    public void ClearLocal()
    {
        CurrentAgent = null;
        string path = Path.Combine(Application.persistentDataPath, BIND_FILE);
        if (File.Exists(path))
            File.Delete(path);
        Debug.Log("[AgentBindModel] 绑定信息已清除");
    }

    /// <summary>检查授权是否过期</summary>
    public bool IsExpired()
    {
        if (CurrentAgent == null) return true;
        if (CurrentAgent.bindMode == 0) return false; // 永久绑定

        if (System.DateTime.TryParse(CurrentAgent.expiredTime, out var expired))
            return System.DateTime.Now > expired;

        return false;
    }
}
```

```csharp
// Assets/Scripts/Game/AgentBindManager.cs
using Newtonsoft.Json;
using System;
using System.Collections;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 代理商绑定逻辑管理器 —— 使用框架 BaseManager 单例模式
/// </summary>
public class AgentBindManager : BaseManager<AgentBindManager>
{
    // 服务器地址 — 从配置读取或硬编码
    private string _serverHost = "https://your-cloud-server.com";
    private const string BIND_API = "/api/agent/bind";
    private const string UNBIND_API = "/api/agent/unbind";
    private const int TIMEOUT = 10;

    /// <summary>
    /// 初始化：读取本地绑定信息
    /// </summary>
    public void Init()
    {
        AgentBindModel.Instance.Init();

        // 监听事件
        LogicMessenger.AddListener(MessageName.Event_AgentBindSuccess, OnBindSuccess);
        LogicMessenger.AddListener(MessageName.Event_AgentUnbind, OnUnbind);
    }

    /// <summary>
    /// ① 执行绑定 → HTTP 请求云端验证
    /// </summary>
    public void DoBind(string bindCode, Action<bool, string> callback = null)
    {
        var req = new AgentBindRequest
        {
            machineCode = SBoxModel.Instance.USN,
            bindCode = bindCode,
            macId = SBoxModel.Instance.macId.ToString()
        };

        string json = JsonConvert.SerializeObject(req);
        Debug.Log($"[AgentBindManager] 发送绑定请求: {bindCode}");

        // 使用框架 Utils.Post 发起 HTTP 请求（与 IoTPayment 中模式一致）
        var param = new Dictionary<string, string>
        {
            { "machineCode", req.machineCode },
            { "bindCode", req.bindCode },
            { "macId", req.macId }
        };

        string respStr = Utils.Post($"{_serverHost}{BIND_API}", param);
        Debug.Log($"[AgentBindManager] 服务器响应: {respStr}");

        try
        {
            var resp = JsonConvert.DeserializeObject<AgentBindResponse>(respStr);

            if (resp != null && resp.code == 0 && resp.data != null)
            {
                // 绑定成功 → 保存到本地
                AgentBindModel.Instance.SaveToLocal(resp.data);

                // 通知 UI
                LogicMessenger.Broadcast(MessageName.Event_AgentBindSuccess);

                callback?.Invoke(true, "绑定成功");
            }
            else
            {
                string errMsg = GetErrorMessage(resp?.code ?? -1, resp?.message);
                LogicMessenger.Broadcast(MessageName.Event_AgentBindFailed);
                callback?.Invoke(false, errMsg);
            }
        }
        catch (Exception e)
        {
            Debug.LogError($"[AgentBindManager] 解析响应异常: {e.Message}");
            callback?.Invoke(false, "服务器响应异常, 请重试");
        }
    }

    /// <summary>
    /// ② 解绑 (管理后台操作)
    /// </summary>
    public void DoUnbind(Action<bool, string> callback = null)
    {
        if (!AgentBindModel.Instance.IsBound)
        {
            callback?.Invoke(false, "机台未绑定");
            return;
        }

        var param = new Dictionary<string, string>
        {
            { "machineCode", SBoxModel.Instance.USN },
            { "agentId", AgentBindModel.Instance.CurrentAgent.agentId }
        };

        string respStr = Utils.Post($"{_serverHost}{UNBIND_API}", param);
        var resp = JsonConvert.DeserializeObject<AgentUnbindResponse>(respStr);

        if (resp != null && resp.code == 0)
        {
            AgentBindModel.Instance.ClearLocal();
            LogicMessenger.Broadcast(MessageName.Event_AgentUnbind);
            callback?.Invoke(true, "解绑成功");
        }
        else
        {
            callback?.Invoke(false, resp?.message ?? "解绑失败");
        }
    }

    // ═══ 事件回调 ═══
    void OnBindSuccess()
    {
        var agent = AgentBindModel.Instance.CurrentAgent;
        Debug.Log($"[AgentBindManager] 绑定成功! 代理商: {agent.agentName}");

        // 后续业务：同步机台配置、上报绑定状态等
    }

    void OnUnbind()
    {
        Debug.Log("[AgentBindManager] 已解绑，返回绑定页面");
    }

    /// <summary>错误码转可读消息</summary>
    string GetErrorMessage(int code, string serverMsg)
    {
        return code switch
        {
            1001 => "绑定码无效",
            1002 => "绑定码已过期",
            1003 => "该机台已被其他代理商绑定",
            1004 => "代理商账号状态异常",
            _ => serverMsg ?? $"验证失败 (code:{code})"
        };
    }
}
```

## 7. 第四步：AgentBindPanel UI面板

```csharp
// Assets/Scripts/Game/AgentBindPanel.cs
using UnityEngine;
using UnityEngine.UI;

/// <summary>
/// 代理商绑定面板 —— 继承框架 BasePanel 获得自动控件绑定
/// 预制体上挂载此脚本即可，BasePanel.Awake() 自动绑定所有子控件
/// </summary>
public class AgentBindPanel : BasePanel
{
    // ═══ UI控件 (通过 BasePanel.GetControl<T>(name) 获取) ═══

    // 显示文本
    private Text _titleText;           // 标题 "代理商绑定"
    private Text _machineCodeText;     // 显示机台编号
    private Text _statusText;          // 状态提示 "请输入16位绑定码"
    private Text _agentInfoText;       // 已绑定时显示代理商名

    // 输入框
    private InputField _bindCodeInput; // 16位绑定码输入

    // 面板
    private GameObject _bindPanel;     // 绑定输入面板
    private GameObject _successPanel;  // 绑定成功面板
    private GameObject _errorPanel;    // 错误提示面板

    // 按钮 (BasePanel 自动绑定 OnClick)
    // "NumButton_0" ~ "NumButton_9" — 数字键
    // "DeleteButton" — 删除键
    // "ClearButton" — 清除键
    // "ConfirmButton" — 确认绑定
    // "ReBindButton" — 重新绑定
    // "ExitButton" — 退出 (仅编辑器)

    private string _bindCode = "";
    private const int BIND_CODE_LENGTH = 16;

    protected override void Awake()
    {
        base.Awake(); // ← BasePanel 自动 FindChildrenControl<Button/Text/InputField/...>

        // 获取控件引用
        _titleText = GetControl<Text>("TitleText");
        _machineCodeText = GetControl<Text>("MachineCodeText");
        _statusText = GetControl<Text>("StatusText");
        _agentInfoText = GetControl<Text>("AgentInfoText");
        _bindCodeInput = GetControl<InputField>("BindCodeInput");

        // 获取面板引用
        _bindPanel = transform.Find("BindPanel")?.gameObject;
        _successPanel = transform.Find("SuccessPanel")?.gameObject;
        _errorPanel = transform.Find("ErrorPanel")?.gameObject;

        // 隐藏所有面板
        ShowPanel("bind");
    }

    void Start()
    {
        // 显示机台编号
        if (_machineCodeText != null)
            _machineCodeText.text = $"机台编号: {SBoxModel.Instance.USN}";

        // 设置输入框为只读（通过虚拟键盘输入）
        if (_bindCodeInput != null)
            _bindCodeInput.readOnly = true;
    }

    // ═══ 按钮回调 (BasePanel 自动调用 OnClick(btnName)) ═══
    protected override void OnClick(string btnName)
    {
        Debug.Log($"[AgentBindPanel] 按钮: {btnName}");

        switch (btnName)
        {
            // ── 数字键 ──
            case "NumButton_0": AppendDigit("0"); break;
            case "NumButton_1": AppendDigit("1"); break;
            case "NumButton_2": AppendDigit("2"); break;
            case "NumButton_3": AppendDigit("3"); break;
            case "NumButton_4": AppendDigit("4"); break;
            case "NumButton_5": AppendDigit("5"); break;
            case "NumButton_6": AppendDigit("6"); break;
            case "NumButton_7": AppendDigit("7"); break;
            case "NumButton_8": AppendDigit("8"); break;
            case "NumButton_9": AppendDigit("9"); break;

            // ── 功能键 ──
            case "DeleteButton":
                DeleteDigit();
                break;
            case "ClearButton":
                ClearBindCode();
                break;
            case "ConfirmButton":
                DoBind();
                break;
            case "ReBindButton":
                ShowPanel("bind");
                break;
            case "ExitButton":
#if UNITY_EDITOR
                // 仅编辑器允许跳过绑定
                LogicMessenger.Broadcast(MessageName.Event_AgentBindSuccess);
#endif
                break;
        }
    }

    // ═══ 数字输入 ═══
    void AppendDigit(string digit)
    {
        if (_bindCode.Length >= BIND_CODE_LENGTH) return;

        _bindCode += digit;
        UpdateCodeDisplay();

        // 输入满16位，自动触发确认
        if (_bindCode.Length == BIND_CODE_LENGTH)
            Invoke(nameof(DoBind), 0.3f);
    }

    void DeleteDigit()
    {
        if (_bindCode.Length == 0) return;
        _bindCode = _bindCode.Substring(0, _bindCode.Length - 1);
        UpdateCodeDisplay();
    }

    void ClearBindCode()
    {
        _bindCode = "";
        UpdateCodeDisplay();

        if (_statusText != null)
            _statusText.text = "请输入16位绑定码";
    }

    void UpdateCodeDisplay()
    {
        if (_bindCodeInput != null)
        {
            // 用 * 遮盖已输入的数字
            _bindCodeInput.text = new string('*', _bindCode.Length)
                + new string('_', BIND_CODE_LENGTH - _bindCode.Length);
        }
    }

    // ═══ 绑定操作 ═══
    void DoBind()
    {
        if (_bindCode.Length != BIND_CODE_LENGTH)
        {
            ShowStatus("请输入完整的16位绑定码", Color.red);
            return;
        }

        ShowStatus("正在验证绑定码...", Color.yellow);

        // 调用 Manager 执行绑定
        AgentBindManager.Instance.DoBind(_bindCode, (success, msg) =>
        {
            if (success)
            {
                ShowPanel("success");
                if (_agentInfoText != null)
                    _agentInfoText.text =
                        $"代理商: {AgentBindModel.Instance.CurrentAgent.agentName}";
            }
            else
            {
                ShowError(msg);
            }
        });
    }

    // ═══ 面板切换 ═══
    void ShowPanel(string panel)
    {
        _bindPanel?.SetActive(panel == "bind");
        _successPanel?.SetActive(panel == "success");
        _errorPanel?.SetActive(panel == "error");

        if (panel == "bind")
            ClearBindCode();
    }

    void ShowStatus(string msg, Color color)
    {
        if (_statusText != null)
        {
            _statusText.text = msg;
            _statusText.color = color;
        }
    }

    void ShowError(string msg)
    {
        ShowStatus(msg, Color.red);
        _errorPanel?.SetActive(true);

        // 3秒后自动隐藏
        CancelInvoke(nameof(HideError));
        Invoke(nameof(HideError), 3f);
    }

    void HideError()
    {
        _errorPanel?.SetActive(false);
        ShowStatus("请输入16位绑定码", Color.white);
    }
}
```

```csharp
// Load.cs — 原有 OnSBoxReadConfigComplete() 末尾添加绑定检查

private void OnSBoxReadConfigComplete()
{
    if (descText != null)
        descText.text = "正在载入...";

    // 设置语言
    IOCanvasModel.Instance.CurLanguage = (Language)(Enum.Parse(
        typeof(Language),

        // 未绑定或已过期 → 显示绑定面板
        if (descText != null)
            descText.text = "等待代理商绑定...";

        LogicMessenger.AddListener(MessageName.Event_AgentBindSuccess, OnAgentBound);

            (obj) =>

            });

    // 已绑定 → 继续原有流程
    ContinueStartup();
}

/// <summary>代理商绑定完成后继续启动</summary>
void OnAgentBound()
{
    LogicMessenger.RemoveListener(MessageName.Event_AgentBindSuccess, OnAgentBound);

    if (descText != null)
        descText.text = "绑定完成，正在载入...";

    ContinueStartup();
}

/// <summary>继续原有启动流程 (原 OnSBoxReadConfigComplete 的逻辑)</summary>
void ContinueStartup()
{
    // 设置网络
    NetMgr.Instance.SetNetAutoConnect(true);

    // 预加载音效
    PreloadSound();

    // 异步加载游戏场景
    ScenesMgr.Instance.LoadSceneAsyn("game", () =>
        AssetBundleMgr.Instance.LoadResAsync<GameObject>("game", "ui", (obj) =>
        {
            Controller.Instance.Init();
        }));

    SBoxIdeaSummary.GetSummary();
}
```

```csharp
public void ShowAgentInfo()
{
    if (!AgentBindModel.Instance.IsBound)
    {
        InstantiateEasyShow("代理商", "未绑定");
        return;
    }

    var agent = AgentBindModel.Instance.CurrentAgent;
    InstantiateEasyShow("代理商", agent.agentName);
    InstantiateEasyShow("绑定时间", agent.bindTime);
    if (agent.bindMode == 1)
        InstantiateEasyShow("到期时间", agent.expiredTime ?? "未知");
    else
        InstantiateEasyShow("授权类型", "永久");
}
```
